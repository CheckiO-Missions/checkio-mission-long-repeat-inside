def repeat_inside(line: str) -> str:
            
    result = []
    for i in range(len(line)):
        for j in range(i+1, len(line) + 1):
            for k in range(2, len(line) + 1):
                if line[i:j] * k in line:
                    result.append(line[i:j] * k)
    return max(result, key= len, default='')

Mit Hilfe des Ausdrucks "re.findall(r'(?=(.+?)\2+))', line)" können wir die Zeichenkette durchgehen und alle Kombinationen ermitteln, die mehr als einmal in der Zeichenkette vorkommen und einfach die längste Teilzeichenkette zurückgeben.

import re


def repeat_inside(line: str) -> str:
            
    result = re.findall(r'(?=((.+?)\2+))', line)
    return max([i[0] for i in result], key= len, default= '')
